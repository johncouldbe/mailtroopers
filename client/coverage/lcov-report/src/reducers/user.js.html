  SET_CURRENT_USER
  } from '../actions/user'
 
const initialState = {
  regErr: null,
  logErr: null,
  authToken: null,
  currentUser: null
}
 
export const userReducer = (state=initialState, action) => {
  if(action.type === REGISTER_ERR){

      regErr: action.err
    })
  }
  if(action.type === LOGIN_ERR){
    return Object.assign({}, state, {
      logErr: action.err
